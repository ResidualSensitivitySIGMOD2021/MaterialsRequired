select nation.N_NATIONKEY, count(nation.N_NATIONKEY) from nation, customer, orders, lineitem, supplier where nation.N_NATIONKEY=customer.C_NATIONKEY and customer.C_CUSTKEY=orders.O_CUSTKEY and orders.O_ORDERKEY=lineitem.L_ORDERKEY and lineitem.L_SUPPKEY=supplier.S_SUPPKEY group by nation.N_NATIONKEY;
