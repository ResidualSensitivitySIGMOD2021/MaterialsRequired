create table edge2_w as select edge2_from, edge2_to, 1 as weight from edge2;

create table edge4_w as select edge4_from, edge4_to, 1 as weight from edge4;
create table edge5_w as select edge5_from, edge5_to, 1 as weight from edge5;
create table keyb_w1 as select edge1_to, sum(weight) as weight from edge1_w group by edge1_to;
create table keyb_w2 as select edge2_from, sum(weight) as weight from edge2_w group by edge2_from;

create table keyc_w as select edge3_from, sum(weight) as weight from edge3_w group by edge3_from;
create table edge123_w as select edge3_to, sum(1.0*edge12_w.weight*edge3_w.weight/(edge12_w.weight+keyc_w.weight)) as weight from  edge12_w, keyc_w, edge3_w where edge12_w.edge2_to=keyc_w.edge3_from and edge12_w.edge2_to=edge3_w.edge3_from group by edge3_to;
create table keyd_w as select edge4_from, sum(weight) as weight from edge4_w group by edge4_from;
create table edge1234_w as select edge4_to, sum(1.0*edge123_w.weight*edge4_w.weight/(edge123_w.weight+keyd_w.weight)) as weight from  edge123_w, keyd_w, edge4_w where edge123_w.edge3_to=keyd_w.edge4_from and edge123_w.edge3_to=edge4_w.edge4_from group by edge4_to;
create table keye_w as select edge5_from, sum(weight) as weight from edge5_w group by edge5_from;
create table edge12345_w as select edge5_to, sum(1.0*edge1234_w.weight*edge5_w.weight/(edge1234_w.weight+keye_w.weight)) as weight from  edge1234_w, keye_w, edge5_w where edge1234_w.edge4_to=keye_w.edge5_from and edge1234_w.edge4_to=edge5_w.edge5_from group by edge5_to;
